   108→// Payment Breakdown Component
   109→function PaymentBreakdown({ order }: { order: any }) {
   110→  const quantity = order.quantity || 0
   111→  const unitPrice = order.unitPrice || order.product?.pricePerUnit || 0
   112→  const productCost = quantity * unitPrice
   113→  const platformFee = order.platformFee || Math.round(productCost * 0.02 * 100) / 100
   114→  const transportBookingFee = order.transportBookingFee || 30
   115→  const isFreeDelivery = order.freeDelivery || order.product?.freeDelivery
   116→  const deliveryFee = order.deliveryFee || 0
   117→  const total = order.totalPayable || (productCost + platformFee + transportBookingFee + (isFreeDelivery ? 0 : deliveryFee))
   118→  const advancePaid = order.advanceAmount || Math.round(total * 0.5 * 100) / 100
   119→  const remaining = order.remainingAmount || Math.round(total * 0.5 * 100) / 100
   120→
   121→  return (
   122→    <motion.div
   123→      initial={{ opacity: 0, height: 0 }}
   124→      animate={{ opacity: 1, height: 'auto' }}
   125→      exit={{ opacity: 0, height: 0 }}
   126→      className="glass-card p-4 border border-amber-500/15 bg-amber-500/[0.03]"
   127→    >
   128→      <div className="flex items-center gap-2 mb-3">
   129→        <CreditCard className="h-4 w-4 text-amber-400" />
   130→        <h5 className="text-sm font-semibold text-foreground">Payment Breakdown</h5>
   131→        {order.paymentStatus && (
   132→          <Badge className={`${paymentStatusColors[order.paymentStatus] || ''} border text-[10px] ml-auto`}>
   133→            {order.paymentStatus === 'advance_paid' ? '50% Paid' : order.paymentStatus === 'full_paid' ? 'Fully Paid' : order.paymentStatus.replace('_', ' ')}
   134→          </Badge>
   135→        )}
   136→      </div>
   137→      <div className="space-y-1.5 text-xs">
   138→        <div className="flex justify-between">
   139→          <span className="text-muted-foreground">Product Cost ({quantity} × ₹{unitPrice.toLocaleString()})</span>
   140→          <span className="text-foreground font-medium">₹{productCost.toLocaleString()}</span>
   141→        </div>
   142→        <div className="flex justify-between">
   143→          <span className="text-muted-foreground">Platform Fee (2%)</span>
   144→          <span className="text-foreground font-medium">₹{platformFee.toLocaleString()}</span>
   145→        </div>
   146→        <div className="flex justify-between">
   147→          <span className="text-muted-foreground">Transport Booking Fee</span>
   148→          <span className="text-foreground font-medium">₹{transportBookingFee.toLocaleString()}</span>
   149→        </div>
   150→        {/* Delivery Fee row — FREE badge or charged amount */}
   151→        {isFreeDelivery ? (
   152→          <div className="flex justify-between items-center">
   153→            <span className="text-muted-foreground">Delivery Fee</span>
   154→            <Badge className="bg-emerald-500/20 text-emerald-400 border-emerald-500/30 border text-[10px]">FREE</Badge>
   155→          </div>
   156→        ) : deliveryFee > 0 ? (
   157→          <div className="flex justify-between">
   158→            <span className="text-muted-foreground">Delivery Fee</span>
   159→            <span className="text-foreground font-medium">₹{deliveryFee.toLocaleString()}</span>
   160→          </div>
   161→        ) : null}
   162→        <div className="flex justify-between pt-1.5 border-t border-glass-border">
   163→          <span className="text-foreground font-semibold">Total</span>
   164→          <span className="text-amber-400 font-bold">₹{total.toLocaleString()}</span>
   165→        </div>
   166→        <div className="flex justify-between pt-1 border-t border-dashed border-glass-border">
   167→          <span className="text-emerald-400">Advance Paid (50%)</span>
   168→          <span className="text-emerald-400 font-medium">₹{advancePaid.toLocaleString()}</span>
   169→        </div>
   170→        <div className="flex justify-between">
   171→          <span className={order.paymentStatus === 'full_paid' ? 'text-emerald-400' : 'text-orange-400'}>
   172→            {order.paymentStatus === 'full_paid' ? 'Remaining (Paid)' : 'Remaining (50%)'}
   173→          </span>
   174→          <span className={order.paymentStatus === 'full_paid' ? 'text-emerald-400 font-medium' : 'text-orange-400 font-bold'}>
   175→            ₹{remaining.toLocaleString()}
   176→          </span>
   177→        </div>
   178→      </div>
   179→    </motion.div>
   180→  )
   181→}
